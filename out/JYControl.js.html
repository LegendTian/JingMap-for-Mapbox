/**
 * 添加绘制控件
 * @method addDrawControl
 * @memberOf Control
 * @example jmap.JYControl.addDrawControl(map,'top-right');
 * @param map=-- {object} 地图对象，不可为空
 * @param position='top-right' {string} 控件位置,'top-left','top-right','bottom-left','bottom-right'之一可为空，默认为‘top-right’

let addDrawControl=function (map,position,point,line,polygon,style) {
    position=position?position:'top-right';
    let draw=null;
    if(style){
        draw=new MapboxDraw({
            displayControlsDefault: true,
            controls: {
                point:point,
                line_string: line,
                polygon: polygon,
                trash: true,
                combine_features :false,
                uncombine_features :false
            },
            styles:style,
            userProperties:false
        });
    }else{
        draw=new MapboxDraw({
            displayControlsDefault: true,
            controls: {
                point:point,
                line_string: line,
                polygon: polygon,
                trash: true,
                combine_features :false,
                uncombine_features :false
            },
            userProperties:false
        });
    }

    map.addControl(draw,position);
    return draw;
};

/**
 * 添加搜索控件
 * @method addSearchControl
 * @memberOf Control
 * @example
 * let search=jmap.JYControl.addSearchControl(map,searchData,19,'top-left');
 * @param map=-- {object} 地图对象，不可为空
 * @param searchData=-- {object} 自定义搜索范围对象，必须为geojson点对象的数组，可为空，默认为全国市、县点
 * @param flyTo=-- {Boolean} 结果是否跳转，可以为空，默认为true
 * @param zoom=-- {number} 搜索结果跳转的缩放级别，可为空，默认为地图当前级别,当flyTo为false时，不可用
 * @param position='top-left' {string} 控件位置,'top-left','top-right','bottom-left','bottom-right'之一可为空，默认为‘top-left’
 * @return control {object} 搜索控件对象
 */
let addSearchControl=function (map,searchData,flyTo,zoom,position) {
    position=position?position:'top-left';
    zoom=zoom?zoom:map.getZoom();
    function forwardGeocoder(query) {
        let matchingFeatures = [];
        for (let i = 0; i < searchData.length; i++) {
            if(searchData[i].features.length){
                searchData[i].features.forEach(function(obj){
                    let feature =obj;
                    Object.keys(obj.properties).forEach(function(key){
                        if (String(obj.properties[key]).toLowerCase().search(query.toLowerCase()) !== -1) {
                            feature['place_name'] =obj.properties[key];
                            feature['center'] = obj.geometry.coordinates;
                            data=obj;
                            matchingFeatures.push(feature);
                        }
                    })
                })

                }

            }
        return matchingFeatures;
    }
    //编辑搜索控件
    let control=new MapboxGeocoder({
        localGeocoder: forwardGeocoder,
        placeholder: "搜索",
        country: "cn",
        language: "zh-Hans",
        flyTo:flyTo,
        zoom: zoom,
        marker:true,
    });
    map.addControl(control, position);
    return control;
};
