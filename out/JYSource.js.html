/**
 * 加载地图源
 * @method addSource
 * @memberOf Source
 * @example
 * let map=jmap.JYMap.addMapboxMap(zzCounty,'mapContainer',9,[112.8547, 35.4992]);
 * map.on('load',function (){
 * //加载DEM数据
 * jmap.JYSource.addSource(map,'raster-dem','zzDem',url3);
 * jmap.JYLayer.addDemLayer(map,'dem图层','zzDem','JYGIS:map_dem','#000','#deff78','#fffa98');
 * //加载栅格数据源
 * jmap.JYSource.addSource(map,'raster','zzraster',url1);
 * jmap.JYLayer.addRasterLayer(map,'栅格图层1','zzraster','JYGIS:map_zezhou_raster',0);
 * //加载矢量数据
 * jmap.JYSource.addSource(map,'vector','zzmap',url2);
 * jmap.JYLayer.addPolygonLayer(map,'town_polygon','zzmap','map_town_polygon','#7ec2ff',1,'#000');
 * //加载视频文件
 * let bounds=[
 * [113.24, 36.38],
 * [112.72, 36.38],
 * [112.72, 35.71],
 * [113.24, 35.71]
 * ];
 * jmap.JYSource.addSource(map,'video','videoSource',url4,bounds);
 * jmap.JYLayer.addMediaLayer(map,'视频1','videoSource');
 * })
 * @param map=-- {object} 地图对象，不可为空
 * @param type=-- {string} 数据源类型，必须为{vector/geojson/raster/raster-dem/video/image/canvas}之一，不可为空
 * @param sourceName=-- {string} 数据源名称，唯一，不可为空
 * @param source=-- {string}  数据源地址url，不可为空
 * @param bound=-- {array} 图像显示边框四角点坐标，当type为{video/image/canvas}之一时，不可为空
 */
let addSource=function (map,type,sourceName,source,bound) {
    if(type==='geojson'&&typeof source==='object'){
        map.addSource(sourceName,{
            type:'geojson',
            data:source
        });
    }    else if(type==='vector'){
        map.addSource(sourceName,{
            type:type,
            scheme:'tms',
            tiles:[source]
        });
    }
    else if(type==='raster'||type==='raster-dem'){
        map.addSource(sourceName,{
            type:type,
            tiles:[source]
        });
    }
    else if(type==='video'||type==='image'||type==='canvas'&&bound){
        map.addSource(sourceName,{
            'type':type,
            'urls':[source],
            'coordinates':bound,
        });
    }else{
        // alert("请输入正确数据源类型！")
    }
};

/**
 * 删除地图源
 * @method removeSource
 * @memberOf Source
 * @example
 * let map=jmap.JYMap.addMapboxMap(zzCounty,'mapContainer',9,[112.8547, 35.4992]);
 * let url2='http://180.76.103.238:8090/geoserver/gwc/service/tms/1.0.0/JYGIS%3Amap1@EPSG%3A3857@pbf/{z}/{x}/{y}.pbf';
 * jmap.JYSource.addSource(map,'vector','zzmap',url2);
 * jmap.JYSource.remaveSource(map,'zzmap');
 * @param map=-- {object} 地图对象，不可为空
 * @param sourceName=-- {string} 数据源名称，唯一，不可为空
 */
let removeSource=function (map,sourceName) {
    map.removeSource(sourceName);

};

module.exports = {
    addSource,
    removeSource
}
