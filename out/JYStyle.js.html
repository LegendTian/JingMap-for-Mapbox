/** @namespace  Style*/
/**
 * 获取地图样式文件
 * @method getMapStyle
 * @memberOf Style
 * @example
 * jmap.JYStyle.getMapStyle(map);
 * @param map=-- {object} 地图对象，不可为空
 * @return style {boject}地图样式对象
 */
let getMapStyle=function (map) {
    return map.getStyle();
};
/**
 * 设置地图样式文件
 * @method setMapStyle
 * @memberOf Style
 * @example
 * jmap.JYStyle.setMapStyle(map,style);
 * @param map=-- {object} 地图对象，不可为空
 * @param style=-- {object} 地图样式对象，不可为空
 * @return style {boject}地图样式对象
 */
let setMapStyle=function (map,style) {
    return map.setStyle(style);
};
/**
 * 获取图层绘制属性
 * @method getLayerPaint

 * @example

 * 获取图层过滤器
 * @method getLayerFilter
 * @memberOf Style
 * @example
 *  jmap.JYStyle.getLayerFilter(map,'road-highlighted')
 * @param map=-- {object} 地图对象，不可为空
 * @param layer=-- {string} 图层ID，不可为空
